from collections import deque

def change_rot(move, letter):
    p, q = move
    if p == 0:
        if letter == "D":
            if q > 0:
                return (1, 0)
            else:
                return (-1, 0)

        else:
            if q > 0:
                return (-1, 0)
            else:
                return (1, 0)
    elif q == 0:
        if letter == "D":
            if p > 0:
                return (0, -1)
            else:
                return (0, 1)
        else:
            if p > 0:
                return (0, 1)
            else:
                return (0, -1)
            
N = int(input())
K = int(input())
board = [[0]*N for _ in range(N)]
for i in range(K):
    x, y = map(int, input().split())
    board[x-1][y-1] = 1
rot = {}
d = int(input())
for i in range(d):
    x, y = map(str, input().split())
    rot[int(x)] = y

board[0][0] = 2
move = (0, 1)
time = 0
p, q = 0, 0
snake = deque()
snake.append((p, q))

while True:
    time += 1
    dp, dq = move
    np, nq = p + dp, q + dq
    if 0 <= np < N and 0 <= nq < N:
        if board[np][nq] == 1:
            snake.appendleft((np, nq))
            board[np][nq] = 2
        elif board[np][nq] == 0:
            board[np][nq] = 2
            snake.appendleft((np, nq))
            fp, fq = snake.pop()
            board[fp][fq] = 0

        elif board[np][nq] == 2:
            break
        p, q = np, nq

        if time in rot:
            move = change_rot(move, rot[time])

    else:
        break

print(time)
